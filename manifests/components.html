    const Wrapper = ({
        containerName = '',

        ...restProps
    }) => {

        const classes = classNames(s.Root, [s[`Root--${name}`]]);

                        <Container {...restProps} />
                    </main>
                </div>

                <Footer {...footer} />
            </div>
        );
    };

        containerName: PropTypes.string,
